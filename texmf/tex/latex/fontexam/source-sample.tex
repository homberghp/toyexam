%% CourseDir is the absolute path to the parent dir of
%% questions/... and builds/.
%% The \ProcessExam at the end assumes the following substructure
%% <CourseDir>
%% ├── builds / (including exampreamble, default and optional
%% |             macros.tex and packages.tex)
%% │   ├── <ExamInstance>, e.g. 20120628
%% │
%% └── questions, and resources e.g.
%%     ├── figures
%%     ├── q1h01
%%     ...
%% when your exam questions input files (such as source code,
%% verbatim files, figures or tables), prefix the file name to be
%% input with "\QuestionBaseDir". Example: in some exams you may have
%% a common figures dir. Including such image into an exam question
%% can be done with
%% \includegraphics[width=140mm]{\QuestionBaseDir/figures/ExcludingGuards}
%% (example from a mod2 exam).
%% If you keep question and figures in the same dir, say state, then
%% adapt the file path accordingly as in
%% \QuestionBaseDir/state/ExcludingGuards.


\providecommand\ProcessorFile{exampreamble}
\def\CourseDir{\string~/modulerepos/databases/exam}

\def\ExamInstance{20140117}

  \EN{January 17$^{th}$ 2014}}
